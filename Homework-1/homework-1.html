2. Anlise dos Fatos Estilizados (Questo 1)

Agora, vamos verificar os fatos estilizados para cada uma das 4 variveis escolhidas.

for (ticker_atual in tickers) {
  
  dados_ativo <- assets %>% filter(ticker == ticker_atual)
  
  # --- Grficos ---
  
  # Grfico dos retornos dirios
  plot_returns <- ggplot(dados_ativo) + geom_line(aes(x = ref.date, y = ret.closing.prices), color = 'darkblue') + labs(x = "", y = 'Retornos', title = paste("Retornos Dirios de", ticker_atual)) + theme_economist()
  
  # Grfico da volatilidade (retornos absolutos)
  plot_volatility <- ggplot(dados_ativo) + geom_line(aes(x = ref.date, y = abs(ret.closing.prices)), color = 'darkred') + labs(x = "", y = 'Retornos Absolutos', title = paste("Volatilidade de", ticker_atual)) + theme_economist()
  
  # Histograma
  plot_histogram <- ggplot(dados_ativo) + geom_histogram(aes(x = ret.closing.prices, y = ..density..), color = "white", fill = "grey50", linetype = "solid", alpha = 0.8) + geom_density(aes(x = ret.closing.prices, y = ..density..), color = "black") + labs(x = "", y = 'Densidade', title = "Histograma dos Retornos") + theme_economist()
  
  # QQ-Plot
  plot_qq <- ggplot(dados_ativo, aes(sample = ret.closing.prices)) + stat_qq() + stat_qq_line() + labs(x = "Quantis Tericos", y = 'Quantis da Amostra', title = "Q-Q Plot") + theme_economist()
  

  # Combinando os grficos em um nico painel
  grid_plots <- cowplot::plot_grid(plot_returns, plot_volatility, plot_histogram, plot_qq, nrow = 2)
  print(grid_plots)
}

4. Coeficiente de Correlao e Grfico de Disperso (Questo Extra 3)

Calculando o coeficiente de correlao entre os retornos de dois ativos (PETR4.SA e VALE3.SA) e gerando o seu respectivo grfico de disperso

# Pivotando os dados para o formato largo
assets_wide <- assets %>%
  select(ref.date, ticker, ret.closing.prices) %>%
  pivot_wider(names_from = ticker, values_from = ret.closing.prices)

# Calculando a correlao entre PETR4 e VALE3
correlation <- cor(assets_wide$`PETR4.SA`, assets_wide$`VALE3.SA`, use = "complete.obs")

# Imprimindo o resultado
print(paste("O coeficiente de correlao entre PETR4.SA e VALE3.SA :", round(correlation, 4)))
## [1] "O coeficiente de correlao entre PETR4.SA e VALE3.SA : 0.4385"
scatter_plot <- ggplot(assets_wide, aes(x = `PETR4.SA`, y = `VALE3.SA`)) +
  geom_point(alpha = 0.5, color = "#006600") +
  geom_smooth(method = "lm", col = "red") +
  labs(
    title = "Grfico de Disperso: Retornos de PETR4 vs. VALE3",
